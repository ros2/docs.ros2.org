169 {
170  std::array<std::array<double, 6>, 6> nested_array = {};
171  size_t l1 = 0, l2 = 0;
172  for (const double & val : row_major) {
173  nested_array[l2][l1] = val;
174 
175  l1++;
176 
177  if (l1 == nested_array[0].size()) {
178  l1 = 0;
179  l2++;
180  }
181  }
182  return nested_array;
183 }
