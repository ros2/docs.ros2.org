42 find_and_replace(
43  const std::basic_string<CharT, Traits, Allocator> & input,
44  const std::basic_string<CharT, Traits, Allocator> & find,
45  const std::basic_string<CharT, Traits, Allocator> & replace)
46 {
47  std::basic_string<CharT, Traits, Allocator> output = input;
48  const std::size_t find_len = find.length();
49  const std::size_t replace_len = replace.length();
50  if (find == replace) {
51  return output;
52  }
53  if (0u == find_len) {
54  return output;
55  }
56  std::size_t pos = output.find(find);
57  while (pos != std::basic_string<CharT, Traits, Allocator>::npos) {
58  output.replace(pos, find_len, replace);
59  pos = output.find(find, pos + replace_len);
60  }
61  return output;
62 }
