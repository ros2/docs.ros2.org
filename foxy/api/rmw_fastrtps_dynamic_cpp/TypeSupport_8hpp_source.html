52 // For C introspection typesupport we create intermediate instances of std::string so that
53 // eprosima::fastcdr::Cdr can handle the string properly.

59  static size_t next_field_align(void * data, size_t current_alignment)
60  {
61  auto c_string = static_cast<rosidl_runtime_c__String *>(data);
62  if (!c_string) {
63  RCUTILS_LOG_ERROR_NAMED(
64  "rmw_fastrtps_dynamic_cpp",
65  "Failed to cast data as rosidl_runtime_c__String");
66  return current_alignment;
67  }
68  if (!c_string->data) {
69  RCUTILS_LOG_ERROR_NAMED(
70  "rmw_fastrtps_dynamic_cpp",
71  "rosidl_generator_c_String had invalid data");
72  return current_alignment;
73  }
74 
75  current_alignment += eprosima::fastcdr::Cdr::alignment(current_alignment, 4);
76  current_alignment += 4;
77  return current_alignment + strlen(c_string->data) + 1;
78  }

80  static std::string convert_to_std_string(void * data)
81  {
82  auto c_string = static_cast<rosidl_runtime_c__String *>(data);
83  if (!c_string) {
84  RCUTILS_LOG_ERROR_NAMED(
85  "rmw_fastrtps_dynamic_cpp",
86  "Failed to cast data as rosidl_runtime_c__String");
87  return "";
88  }
89  if (!c_string->data) {
90  RCUTILS_LOG_ERROR_NAMED(
91  "rmw_fastrtps_dynamic_cpp",
92  "rosidl_generator_c_String had invalid data");
93  return "";
94  }
95  return std::string(c_string->data);
96  }
