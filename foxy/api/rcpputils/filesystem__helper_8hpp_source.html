422 inline bool create_directories(const path & p)
423 {
424  path p_built;
425  int status = 0;
426 
427  for (auto it = p.cbegin(); it != p.cend() && status == 0; ++it) {
428  if (!p_built.empty() || it->empty()) {
429  p_built /= *it;
430  } else {
431  p_built = *it;
432  }
433  if (!p_built.exists()) {
434 #ifdef _WIN32
435  status = _mkdir(p_built.string().c_str());
436 #else
437  status = mkdir(p_built.string().c_str(), S_IRWXU | S_IRWXG | S_IRWXO);
438 #endif
439  }
440  }
441  return status == 0;
442 }

450 inline bool remove(const path & p)
451 {
452 #ifdef _WIN32
453  struct _stat s;
454  if (_stat(p.string().c_str(), &s) == 0) {
455  if (s.st_mode & S_IFDIR) {
456  return _rmdir(p.string().c_str()) == 0;
457  }
458  if (s.st_mode & S_IFREG) {
459  return ::remove(p.string().c_str()) == 0;
460  }
461  }
462  return false;
463 #else
464  return ::remove(p.string().c_str()) == 0;
465 #endif

477 inline path remove_extension(const path & file_path, int n_times = 1)
478 {
479  path new_path(file_path);
480  for (int i = 0; i < n_times; i++) {
481  const auto new_path_str = new_path.string();
482  const auto last_dot = new_path_str.find_last_of('.');
483  if (last_dot == std::string::npos) {
484  return new_path;
485  }
486  new_path = path(new_path_str.substr(0, last_dot));
487  }
488  return new_path;
489 }
