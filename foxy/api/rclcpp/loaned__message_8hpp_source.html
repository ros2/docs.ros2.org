58  LoanedMessage(

72  rclcpp::exceptions::throw_from_rcl_error(ret);

74  message_ = static_cast<MessageT *>(message_ptr);
75  } else {
76  RCLCPP_INFO_ONCE(

78  "Currently used middleware can't loan messages. Local allocator will be used.");
