96  template<typename MessageT>
97  rcl_subscription_options_t
98  to_rcl_subscription_options(const rclcpp::QoS & qos) const
99  {
100  rcl_subscription_options_t result = rcl_subscription_get_default_options();
101  using AllocatorTraits = std::allocator_traits<Allocator>;
102  using MessageAllocatorT = typename AllocatorTraits::template rebind_alloc<MessageT>;
103  auto message_alloc = std::make_shared<MessageAllocatorT>(*allocator.get());
104  result.allocator = allocator::get_rcl_allocator<MessageT>(*message_alloc);
105  result.qos = qos.get_rmw_qos_profile();
106  result.rmw_subscription_options.ignore_local_publications = this->ignore_local_publications;
107 
108  // Apply payload to rcl_subscription_options if necessary.
109  if (rmw_implementation_payload && rmw_implementation_payload->has_been_customized()) {
110  rmw_implementation_payload->modify_rmw_subscription_options(result.rmw_subscription_options);
111  }
112 
113  return result;
114  }

123  return this->allocator;
124  }
