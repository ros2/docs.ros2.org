133  template<typename ResponseT, typename TimeRepT = int64_t, typename TimeT = std::milli>
134  rclcpp::FutureReturnCode
135  spin_until_future_complete(
136  std::shared_future<ResponseT> & future,
137  std::chrono::duration<TimeRepT, TimeT> timeout = std::chrono::duration<TimeRepT, TimeT>(-1))
138  {
139  std::future_status status = future.wait_for(std::chrono::seconds(0));
140  if (status == std::future_status::ready) {
141  return rclcpp::FutureReturnCode::SUCCESS;
142  }
143 
144  auto end_time = std::chrono::steady_clock::now();
145  std::chrono::nanoseconds timeout_ns = std::chrono::duration_cast<std::chrono::nanoseconds>(
146  timeout);
147  if (timeout_ns > std::chrono::nanoseconds::zero()) {
148  end_time += timeout_ns;
149  }
150  std::chrono::nanoseconds timeout_left = timeout_ns;
151 
152  entities_collector_ = std::make_shared<StaticExecutorEntitiesCollector>();
153  entities_collector_->init(&wait_set_, memory_strategy_, &interrupt_guard_condition_);
154 
155  while (rclcpp::ok(this->context_)) {
156  // Do one set of work.
157  entities_collector_->refresh_wait_set(timeout_left);
158  execute_ready_executables();
159  // Check if the future is set, return SUCCESS if it is.
160  status = future.wait_for(std::chrono::seconds(0));
161  if (status == std::future_status::ready) {
162  return rclcpp::FutureReturnCode::SUCCESS;
163  }
164  // If the original timeout is < 0, then this is blocking, never TIMEOUT.
165  if (timeout_ns < std::chrono::nanoseconds::zero()) {
166  continue;
167  }
168  // Otherwise check if we still have time to wait, return TIMEOUT if not.
169  auto now = std::chrono::steady_clock::now();
170  if (now >= end_time) {
171  return rclcpp::FutureReturnCode::TIMEOUT;
172  }
173  // Subtract the elapsed time from the original timeout.
174  timeout_left = std::chrono::duration_cast<std::chrono::nanoseconds>(end_time - now);
175  }
176 
177  // The future did not complete before ok() returned false, return INTERRUPTED.
178  return rclcpp::FutureReturnCode::INTERRUPTED;
179  }
