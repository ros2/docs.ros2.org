74  void
75  onNewDataMessage(eprosima::fastrtps::Subscriber * sub)
76  {
77  assert(sub);
78 
79  CustomClientResponse response;
80  // Todo(sloretz) eliminate heap allocation pending eprosima/Fast-CDR#19
81  response.buffer_.reset(new eprosima::fastcdr::FastBuffer());
82 
83  rmw_fastrtps_shared_cpp::SerializedData data;
84  data.is_cdr_buffer = true;
85  data.data = response.buffer_.get();
86  data.impl = nullptr; // not used when is_cdr_buffer is true
87  if (sub->takeNextData(&data, &response.sample_info_)) {
88  if (eprosima::fastrtps::rtps::ALIVE == response.sample_info_.sampleKind) {
89  response.sample_identity_ = response.sample_info_.related_sample_identity;
90 
91  if (response.sample_identity_.writer_guid() == info_->writer_guid_) {
92  std::lock_guard<std::mutex> lock(internalMutex_);
93 
94  if (conditionMutex_ != nullptr) {
95  std::unique_lock<std::mutex> clock(*conditionMutex_);
96  list.emplace_back(std::move(response));
97  // the change to list_has_data_ needs to be mutually exclusive with
98  // rmw_wait() which checks hasData() and decides if wait() needs to
99  // be called
100  list_has_data_.store(true);
101  clock.unlock();
102  conditionVariable_->notify_one();
103  } else {
104  list.emplace_back(std::move(response));
105  list_has_data_.store(true);
106  }
107  }
108  }
109  }
110  }

146  void onSubscriptionMatched(
147  eprosima::fastrtps::Subscriber * sub,
148  eprosima::fastrtps::rtps::MatchingInfo & matchingInfo)
149  {
150  (void)sub;
151  if (info_ == nullptr) {
152  return;
153  }
154  if (eprosima::fastrtps::rtps::MATCHED_MATCHING == matchingInfo.status) {
155  publishers_.insert(matchingInfo.remoteEndpointGuid);
156  } else if (eprosima::fastrtps::rtps::REMOVED_MATCHING == matchingInfo.status) {
157  publishers_.erase(matchingInfo.remoteEndpointGuid);
158  } else {
159  return;
160  }
161  info_->response_subscriber_matched_count_.store(publishers_.size());
162  }
163 
164 private:
165  bool popResponse(CustomClientResponse & response) RCPPUTILS_TSA_REQUIRES(internalMutex_)
166  {
167  if (!list.empty()) {
168  response = std::move(list.front());
169  list.pop_front();
170  list_has_data_.store(!list.empty());
171  return true;
172  }
173  return false;
174  };

182  std::set<eprosima::fastrtps::rtps::GUID_t> publishers_;
183 };
