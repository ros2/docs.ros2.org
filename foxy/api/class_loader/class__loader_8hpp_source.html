235  template<class Base>
236  void onPluginDeletion(Base * obj)
237  {
238  CONSOLE_BRIDGE_logDebug(
239  "class_loader::ClassLoader: Calling onPluginDeletion() for obj ptr = %p.\n",
240  reinterpret_cast<void *>(obj));
241  if (nullptr == obj) {
242  return;
243  }
244  std::lock_guard<std::recursive_mutex> lock(plugin_ref_count_mutex_);
245  delete (obj);
246  assert(plugin_ref_count_ > 0);
247  --plugin_ref_count_;
248  if (plugin_ref_count_ == 0 && isOnDemandLoadUnloadEnabled()) {
249  if (!ClassLoader::hasUnmanagedInstanceBeenCreated()) {
250  unloadLibraryInternal(false);
251  } else {
252  CONSOLE_BRIDGE_logWarn(
253  "class_loader::ClassLoader: "
254  "Cannot unload library %s even though last shared pointer went out of scope. "
255  "This is because createUnmanagedInstance was used within the scope of this process, "
256  "perhaps by a different ClassLoader. "
257  "Library will NOT be closed.",
258  getLibraryPath().c_str());
259  }
260  }
261  }

276  template<class Base>
277  Base * createRawInstance(const std::string & derived_class_name, bool managed)
278  {
279  if (!managed) {
280  this->setUnmanagedInstanceBeenCreated(true);
281  }
282 
283  if (
284  managed &&
285  ClassLoader::hasUnmanagedInstanceBeenCreated() &&
286  isOnDemandLoadUnloadEnabled())
287  {
288  CONSOLE_BRIDGE_logInform(
289  "%s",
290  "class_loader::ClassLoader: "
291  "An attempt is being made to create a managed plugin instance (i.e. boost::shared_ptr), "
292  "however an unmanaged instance was created within this process address space. "
293  "This means libraries for the managed instances will not be shutdown automatically on "
294  "final plugin destruction if on demand (lazy) loading/unloading mode is used."
295  );
296  }
297  if (!isLibraryLoaded()) {
298  loadLibrary();
299  }
300 
301  Base * obj = class_loader::impl::createInstance<Base>(derived_class_name, this);
302  assert(obj != NULL); // Unreachable assertion if createInstance() throws on failure.
303 
304  if (managed) {
305  std::lock_guard<std::recursive_mutex> lock(plugin_ref_count_mutex_);
306  ++plugin_ref_count_;
307  }
308 
309  return obj;
310  }
