325  void add(const MEvent & evt)
326  {
327  if (target_frames_.empty()) {
328  return;
329  }
330 
331  namespace mt = message_filters::message_traits;
332  const MConstPtr & message = evt.getMessage();
333  std::string frame_id = stripSlash(mt::FrameId<M>::value(*message));
334  rclcpp::Time stamp = mt::TimeStamp<M>::value(*message);
335 
336  if (frame_id.empty()) {
337  messageDropped(evt, filter_failure_reasons::EmptyFrameID);
338  return;
339  }
340 
341  // iterate through the target frames and add requests for each of them
342  MessageInfo info;
343  info.handles.reserve(expected_success_count_);
344  {
345  V_string target_frames_copy;
346  // Copy target_frames_ to avoid deadlock from #79
347  {
348  std::unique_lock<std::mutex> frames_lock(target_frames_mutex_);
349  target_frames_copy = target_frames_;
350  }
351 
352  V_string::iterator it = target_frames_copy.begin();
353  V_string::iterator end = target_frames_copy.end();
354  for (; it != end; ++it) {
355  const std::string & target_frame = *it;
356  auto future = buffer_.waitForTransform(
357  target_frame,
358  frame_id,
359  tf2::timeFromSec(stamp.seconds()),
360  buffer_timeout_,
361  std::bind(&MessageFilter::transformReadyCallback, this, std::placeholders::_1, next_handle_index_));
362 
363  try {
364  const auto status = future.wait_for(std::chrono::seconds(0));
365  if (status == std::future_status::ready) {
366  future.get();
367  // Transform is available
368  ++info.success_count;
369  }
370  else {
371  info.handles.push_back(next_handle_index_++);
372  }
373  }
374  catch (const std::exception & e) {
375  TF2_ROS_MESSAGEFILTER_WARN("Message dropped because: %s", e.what());
376  messageDropped(evt, filter_failure_reasons::OutTheBack);
377  return;
378  }
379 
380  if (time_tolerance_.nanoseconds()) {
381  future = buffer_.waitForTransform(
382  target_frame,
383  frame_id,
384  tf2::timeFromSec((stamp + time_tolerance_).seconds()),
385  buffer_timeout_,
386  std::bind(&MessageFilter::transformReadyCallback, this, std::placeholders::_1, next_handle_index_));
387  try {
388  const auto status = future.wait_for(std::chrono::seconds(0));
389  if (status == std::future_status::ready) {
390  future.get();
391  // Transform is available
392  ++info.success_count;
393  }
394  else {
395  info.handles.push_back(next_handle_index_++);
396  }
397  }
398  catch (const std::exception & e) {
399  TF2_ROS_MESSAGEFILTER_WARN("Message dropped because: %s", e.what());
400  messageDropped(evt, filter_failure_reasons::OutTheBack);
401  return;
402  }
403  }
404  }
405  }
406 
407 
408  // We can transform already
409  if (info.success_count == expected_success_count_) {
410  messageReady(evt);
411  } else {
412  // If this message is about to push us past our queue size, erase the oldest message
413  if (queue_size_ != 0 && message_count_ + 1 > queue_size_) {
414 
415  // While we're using the reference keep a lock on the messages.
416  std::unique_lock<std::mutex> unique_lock(messages_mutex_);
417 
418  ++dropped_message_count_;
419  const MessageInfo & front = messages_.front();
420  TF2_ROS_MESSAGEFILTER_DEBUG(
421  "Removed oldest message because buffer is full, count now %d (frame_id=%s, stamp=%f)",
422  message_count_,
423  (mt::FrameId<M>::value(*front.event.getMessage())).c_str(),
424  mt::TimeStamp<M>::value(*front.event.getMessage()).seconds());
425 
426  messageDropped(front.event, filter_failure_reasons::Unknown);
427 
428  messages_.pop_front();
429  --message_count_;
430  }
431 
432  // Add the message to our list
433  info.event = evt;
434  messages_.push_back(info);
435  ++message_count_;
436  }
437 
438  TF2_ROS_MESSAGEFILTER_DEBUG("Added message in frame %s at time %.3f, count now %d",
439  frame_id.c_str(), stamp.seconds(), message_count_);
440  ++incoming_message_count_;
441  }

490  void transformReadyCallback(const tf2_ros::TransformStampedFuture& future, const uint64_t handle)
491  {
492  namespace mt = message_filters::message_traits;
493 
494  // find the message this request is associated with
495  typename L_MessageInfo::iterator msg_it = messages_.begin();
496  typename L_MessageInfo::iterator msg_end = messages_.end();
497  for (; msg_it != msg_end; ++msg_it) {
498  MessageInfo & info = *msg_it;
499  auto handle_it = std::find(info.handles.begin(), info.handles.end(), handle);
500  if (handle_it != info.handles.end()) {
501  // found msg_it
502  ++info.success_count;
503  break;
504  }
505  }
506 
507  if (msg_it == msg_end) {
508  return;
509  }
510 
511  const MessageInfo & info = *msg_it;
512  if (info.success_count < expected_success_count_) {
513  return;
514  }
515 
516  bool can_transform = true;
517  const MConstPtr & message = info.event.getMessage();
518  std::string frame_id = stripSlash(mt::FrameId<M>::value(*message));
519  rclcpp::Time stamp = mt::TimeStamp<M>::value(*message);
520 
521  bool transform_available = true;
522  try {
523  future.get();
524  } catch (...)
525  {
526  transform_available = false;
527  }
528 
529  if (transform_available) {
530  std::unique_lock<std::mutex> frames_lock(target_frames_mutex_);
531  // make sure we can still perform all the necessary transforms
532  typename V_string::iterator it = target_frames_.begin();
533  typename V_string::iterator end = target_frames_.end();
534  for (; it != end; ++it) {
535  const std::string & target = *it;
536  if (!buffer_.canTransform(target, frame_id, tf2::timeFromSec(stamp.seconds()), NULL)) {
537  can_transform = false;
538  break;
539  }
540 
541  if (time_tolerance_.nanoseconds()) {
542  if (!buffer_.canTransform(target, frame_id,
543  tf2::timeFromSec((stamp + time_tolerance_).seconds()), NULL))
544  {
545  can_transform = false;
546  break;
547  }
548  }
549  }
550  } else {
551  can_transform = false;
552  }
553 
554  // We will be mutating messages now, require unique lock
555  std::unique_lock<std::mutex> lock(messages_mutex_);
556  if (can_transform) {
557  TF2_ROS_MESSAGEFILTER_DEBUG("Message ready in frame %s at time %.3f, count now %d",
558  frame_id.c_str(), stamp.seconds(), message_count_ - 1);
559 
560  ++successful_transform_count_;
561  messageReady(info.event);
562  } else {
563  ++dropped_message_count_;
564 
565  TF2_ROS_MESSAGEFILTER_DEBUG("Discarding message in frame %s at time %.3f, count now %d",
566  frame_id.c_str(), stamp.seconds(), message_count_ - 1);
567  messageDropped(info.event, filter_failure_reasons::Unknown);
568  }
569 
570  messages_.erase(msg_it);
571  --message_count_;
572  }

582  void checkFailures()
583  {
584  if (!next_failure_warning_.nanoseconds()) {
585  next_failure_warning_ = node_clock_->get_clock()->now() + rclcpp::Duration(15, 0);
586  }
587 
588  if (node_clock_->get_clock()->now() >= next_failure_warning_) {
589  if (incoming_message_count_ - message_count_ == 0) {
590  return;
591  }
592 
593  double dropped_pct = static_cast<double>(dropped_message_count_) /
594  static_cast<double>(incoming_message_count_ - message_count_);
595  if (dropped_pct > 0.95) {
596  TF2_ROS_MESSAGEFILTER_WARN(
597  "Dropped %.2f%% of messages so far. Please turn the [%s.message_notifier] rosconsole logger to DEBUG for more information.", dropped_pct * 100,
598  "tf2_ros_message_filter");
599  next_failure_warning_ = node_clock_->get_clock()->now() + rclcpp::Duration(60, 0);
600 
601  if (static_cast<double>(failed_out_the_back_count_) / static_cast<double>(dropped_message_count_) > 0.5) {
602  TF2_ROS_MESSAGEFILTER_WARN(
603  " The majority of dropped messages were due to messages growing older than the TF cache time. The last message's timestamp was: %f, and the last frame_id was: %s",
604  last_out_the_back_stamp_.seconds(), last_out_the_back_frame_.c_str());
605  }
606  }
