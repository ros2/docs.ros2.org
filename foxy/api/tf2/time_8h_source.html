57  inline Duration durationFromSec(double t_sec)
58  {
59  int32_t sec, nsec;
60  sec = static_cast<int32_t>(floor(t_sec));
61  nsec = static_cast<int32_t>(std::round((t_sec - sec) * 1e9));
62  // avoid rounding errors
63  sec += (nsec / 1000000000l);
64  nsec %= 1000000000l;
65  return std::chrono::seconds(sec) + std::chrono::nanoseconds(nsec);
66  }
